loop

a simple loop :

loop : (s -> bool) -> (s -> s) -> (s -> s)

loop cond iter actual =

if (cond actual)

then loop iter cond (iter actual)

else actual

a monadic version [the iteration may have effect] :

loop : Monad m => (s -> bool) -> (s -> m s) -> (s -> m s)

loop cond iter actual =

if (cond actual)

then loop iter cond =<< iter actual

else return actual
